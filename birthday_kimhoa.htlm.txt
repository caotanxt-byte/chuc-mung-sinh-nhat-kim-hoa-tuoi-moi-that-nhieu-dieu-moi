<script>
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    let width, height;
    let particles = [];
    let isBlownOut = false;
    let audioContext, analyser, microphone, dataArray;
    let blowIntensity = 0;

    function resize() {
        width = canvas.width = window.innerWidth;
        height = canvas.height = window.innerHeight;
    }
    window.addEventListener('resize', resize);
    resize();

    // Hệ thống âm thanh (Mic)
    async function initAudio() {
        if (audioContext) return; // Đã khởi tạo
        try {
            const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
            audioContext = new (window.AudioContext || window.webkitAudioContext)();
            analyser = audioContext.createAnalyser();
            microphone = audioContext.createMediaStreamSource(stream);
            microphone.connect(analyser);
            analyser.fftSize = 256;
            dataArray = new Uint8Array(analyser.frequencyBinCount);
            document.getElementById('guide').innerText = "HÃY THỔI VÀO MIC...";
        } catch (err) {
            console.log("Mic error:", err);
            document.getElementById('guide').innerText = "LỖI MIC - CLICK ĐỂ TẮT NẾN";
        }
    }

    // Lắng nghe Click như một cách dự phòng
    window.addEventListener('click', (e) => {
        if (e.target.id === 'guide') return; // Bỏ qua click vào nút guide
        if (!isBlownOut) triggerBlowOut();
    });

    // Lớp Hạt cho ngọn lửa và vụ nổ bụi vàng
    class Particle {
        constructor(x, y, isFire = true) {
            this.isFire = isFire;
            this.reset(x, y);
        }
        reset(x, y) {
            this.x = x + (Math.random() - 0.5) * 10;
            this.y = y;
            if (this.isFire) {
                this.size = Math.random() * 15 + 10;
                this.speedY = Math.random() * -3 - 1;
                this.speedX = (Math.random() - 0.5) * 1;
                this.life = Math.random() * 0.5 + 0.5;
                this.color = `rgba(255, ${Math.floor(Math.random() * 100 + 100)}, 0, `;
            } else {
                this.size = Math.random() * 4 + 1;
                let angle = Math.random() * Math.PI * 2;
                let force = Math.random() * 10 + 2;
                this.speedX = Math.cos(angle) * force;
                this.speedY = Math.sin(angle) * force - 2;
                this.life = 1;
                this.decay = Math.random() * 0.02 + 0.01;
                this.color = Math.random() > 0.5 ? '212, 175, 55' : '255, 255, 255'; // Vàng & Trắng
            }
        }
        update() {
            if (this.isFire) {
                // Lửa bị thổi nghiêng
                this.x += this.speedX + (blowIntensity * 5); 
                this.y += this.speedY;
                this.size -= 0.5;
                if (this.size <= 0) this.reset(width / 2, height / 2 + 80);
            } else {
                // Bụi bay
                this.x += this.speedX;
                this.y += this.speedY;
                this.speedY += 0.1; // Trọng lực
                this.speedX *= 0.96; // Lực cản
                this.life -= this.decay;
            }
        }
        draw() {
            if (this.size <= 0 || this.life <= 0) return;
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
            if (this.isFire) {
                ctx.fillStyle = this.color + (this.size / 25) + ')';
            } else {
                ctx.fillStyle = `rgba(${this.color}, ${this.life})`;
                ctx.shadowBlur = 10;
                ctx.shadowColor = `rgba(${this.color}, ${this.life})`;
            }
            ctx.fill();
            ctx.shadowBlur = 0; // Reset
        }
    }

    // Khởi tạo hạt lửa
    for (let i = 0; i < 150; i++) {
        particles.push(new Particle(width / 2, height / 2 + 80, true));
    }

    function triggerBlowOut() {
        isBlownOut = true;
        document.getElementById('guide').style.opacity = 0;
        
        // Chuyển hạt lửa thành bụi nổ
        let cx = width / 2;
        let cy = height / 2 + 80;
        particles = [];
        for (let i = 0; i < 300; i++) {
            particles.push(new Particle(cx, cy, false));
        }

        // Hiện chữ điện ảnh
        setTimeout(() => document.getElementById('t0').classList.add('show'), 500);
        setTimeout(() => document.getElementById('t1').classList.add('show'), 1500);
        setTimeout(() => document.getElementById('t2').classList.add('show'), 2500);
        setTimeout(() => document.getElementById('t3').classList.add('show'), 3500);
        setTimeout(() => document.getElementById('t4').classList.add('show'), 4800);
    }

    function drawCandle(cx, cy) {
        // Ánh sáng tỏa ra từ nến (Glow)
        let glow = ctx.createRadialGradient(cx, cy - 20, 0, cx, cy - 20, 200 + blowIntensity * 100);
        glow.addColorStop(0, 'rgba(255, 150, 0, 0.15)');
        glow.addColorStop(1, 'rgba(0, 0, 0, 0)');
        ctx.fillStyle = glow;
        ctx.fillRect(0, 0, width, height);

        // Thân nến 3D (Gradient)
        let candleGrad = ctx.createLinearGradient(cx - 15, 0, cx + 15, 0);
        candleGrad.addColorStop(0, '#5a1818');
        candleGrad.addColorStop(0.5, '#b43232');
        candleGrad.addColorStop(1, '#3c0a0a');
        
        ctx.fillStyle = candleGrad;
        ctx.fillRect(cx - 15, cy, 30, 150);
        
        // Miệng nến
        ctx.beginPath();
        ctx.ellipse(cx, cy, 15, 5, 0, 0, Math.PI * 2);
        ctx.fillStyle = '#8c2828';
        ctx.fill();

        // Tim nến
        ctx.fillStyle = '#111';
        ctx.fillRect(cx - 2, cy - 10, 4, 10);
    }

    function animate() {
        ctx.fillStyle = 'rgba(3, 3, 5, 0.3)'; // Tạo hiệu ứng vệt mờ (motion blur)
        ctx.fillRect(0, 0, width, height);
        
        // Xử lý Mic
        if (analyser && !isBlownOut) {
            analyser.getByteFrequencyData(dataArray);
            let sum = 0;
            for(let i = 0; i < dataArray.length; i++) { sum += dataArray[i]; }
            let average = sum / dataArray.length;
            
            blowIntensity = average / 100; // Đo cường độ
            if (blowIntensity > 0.8) { // Cường độ đủ lớn -> Tắt nến
                triggerBlowOut();
            }
        }

        if (!isBlownOut) {
            ctx.globalCompositeOperation = 'screen'; // Hòa trộn ánh sáng
            drawCandle(width / 2, height / 2 + 80);
            particles.forEach(p => { p.update(); p.draw(); });
            ctx.globalCompositeOperation = 'source-over';
        } else {
            // Rút nến xuống từ từ
            ctx.globalCompositeOperation = 'lighter';
            particles.forEach(p => { p.update(); p.draw(); });
            ctx.globalCompositeOperation = 'source-over';
        }

        requestAnimationFrame(animate);
    }

    animate();
</script>
